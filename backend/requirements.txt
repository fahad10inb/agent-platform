# Part 1 dependencies — kept tiny on purpose. We add more as we need them.
#
# fastapi  : the web framework — lets us define "routes" (URLs the server answers).
# uvicorn  : the server that actually runs our app and listens for requests.
#            [standard] pulls in a few helpers (auto-reload, faster networking).
# pydantic-settings : reads our config from environment variables / a .env file.

fastapi>=0.115,<1
uvicorn[standard]>=0.30,<1
pydantic-settings>=2.4,<3

# google-genai : the official Gemini SDK (the AI brain). We use its async client.
google-genai>=2.0,<3

# psycopg : the Postgres driver — lets db.py talk to Supabase. [binary] ships a
# precompiled build so there's nothing to compile locally; [pool] adds the
# shared connection pool (no per-query TLS handshake).
psycopg[binary,pool]>=3.1,<4

# tzdata : timezone database so zoneinfo("Asia/Dubai") works on any host —
# the agent must think in Gulf time, not the server's UTC.
tzdata>=2024.1

# httpx : used directly by the WhatsApp channel (app/whatsapp.py) to POST
# replies to Meta's Graph API. It arrives transitively via google-genai today,
# but the WhatsApp channel must not silently break if that transitive pin
# changes — so we declare it explicitly.
httpx>=0.28,<1

# sentry-sdk : error monitoring. Completely inert unless SENTRY_DSN is set, so
# it changes nothing until you paste a DSN — then unhandled 500s get reported
# with the request context instead of vanishing into the logs.
sentry-sdk>=2,<3
